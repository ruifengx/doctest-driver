{-# LANGUAGE CPP #-}

module GHC.Paths (
        ghc, ghc_pkg, libdir, docdir
  ) where

libdir, docdir, ghc, ghc_pkg :: FilePath

libdir :: FilePath
libdir  = GHC_PATHS_LIBDIR
docdir :: FilePath
docdir  = GHC_PATHS_DOCDIR

ghc :: FilePath
ghc     = GHC_PATHS_GHC
ghc_pkg :: FilePath
ghc_pkg = GHC_PATHS_GHC_PKG
